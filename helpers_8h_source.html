135 static inline struct evmc_result evmc_make_result(enum evmc_status_code status_code,
136  int64_t gas_left,
137  const uint8_t* output_data,
138  size_t output_size)
139 {
140  struct evmc_result result;
141  memset(&result, 0, sizeof(result));
142 
143  if (output_size != 0)
144  {
145  uint8_t* buffer = (uint8_t*)malloc(output_size);
146 
147  if (!buffer)
148  {
149  result.status_code = EVMC_OUT_OF_MEMORY;
150  return result;
151  }
152 
153  memcpy(buffer, output_data, output_size);
154  result.output_data = buffer;
155  result.output_size = output_size;
156  result.release = evmc_free_result_memory;
157  }
158 
159  result.status_code = status_code;
160  result.gas_left = gas_left;
161  return result;
162 }
